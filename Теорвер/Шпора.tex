\section{Вступительная лекция}
\begin{notation} \ \\
\begin{itemize}
\item $\Omega$ - пространство элементарных исходов.
\item $\#(A)$ - число благополучных исходов. 
\item $S_n$ - число успехов.
\item $\nu_n = \dfrac{S_n}{n}$ - частота (доля успехов).
\item $P(A)$ - вероятность.
\item $\Lambda_k(A)$ - $k$-мерная мера Лебега.
\end{itemize}
\end{notation}
\subsection{Модель классической вероятности}
\begin{enumerate}
    \item $\Omega$ - конечно.
    \item $\omega_i$ - равновозможны (симметричны).
    \item $\forall A \subseteq \Omega, \  P(A) = \dfrac{\#(A)}{\#(\Omega)}$ 
\end{enumerate}
\subsection{Модель дискретной вероятности}
\begin{enumerate}
    \item $\Omega$ - не более, чем счетно.
    \item $\sum\limits_{i,  \omega_i \in \Omega} p_i = 1$
    \item $P(A) = \sum \limits_{i, \omega_i \in A} p_i$
\end{enumerate}
\subsection{Модель геометрической вероятности}
\begin{enumerate}
    \item $\Omega$ - ограниченная область в $\mathbb{R}^n$.
    \item $P(a) = \dfrac{\Lambda_n(A)}{\Lambda_n (\Omega)} $
\end{enumerate}
